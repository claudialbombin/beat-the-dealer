# NumPy: Fundamental package for scientific computing in Python.
# Used for: Array operations, statistical calculations (mean, std),
#           random number generation, and efficient numerical computations
#           throughout the Monte Carlo simulation engine.
numpy>=1.21.0

# ---------------------------------------------------------------------------
# Data Visualization
# ---------------------------------------------------------------------------

# Matplotlib: Comprehensive plotting library for Python.
# Used for: Generating all project visualizations including
#           strategy heatmaps, EV vs True Count plots (THE KEY GRAPH),

#           and the summary dashboard.
# Version 3.5.0+ required for seaborn-v0_8-darkgrid style support.
matplotlib>=3.5.0

# Seaborn: Statistical data visualization built on matplotlib.
# Used for: Enhanced visual styling, color palettes,
#           and default plot aesthetics that make the
#           visualizations look professional and polished.
seaborn>=0.11.0

# ---------------------------------------------------------------------------
# Progress Tracking
# ---------------------------------------------------------------------------

# tqdm: Fast, extensible progress bar for loops.

#           - Card counting simulation (hundreds of shoes)

tqdm>=4.62.0
